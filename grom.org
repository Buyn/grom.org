(defn form-group-item
  [{:keys [id type value values]}]
  (let [chenge-value-to-atom
          #(swap! values
                  assoc
                    (keyword id)
                    (.. % -target -value))]
    [:div.form__group
   [:label.form__label {:for id} id]
   [:input.form__input {:type type
                        :id id
                        :value value
                        :on-change chenge-value-to-atom}]]))

(defn form-sold-out
  [values]
  (let [toggle-checkbox

(defn gig-editor-toggle
  ([app] (swap! app update-in [:gig-editor :modal] not))
  ([app status]
         (swap! app assoc-in [:gig-editor :modal] status)))


(defn gig-editor-open
  ([app]
   (gig-editor-toggle app true))
  ([app {:keys [id title desc price img sold-out] :as el}]
    (swap! app
          assoc-in [:gig-editor :values]
          { :id       (or id (str "gig-" (random-uuid)))
            :title    (str/trim title)
            :desc     (str/trim desc)
            :img      (str/trim img)
            :price    (js/parseInt price)
            :sold-out sold-out})
   (gig-editor-toggle app true)))

(defn gig-editor-new
  [app]
  (gig-editor-open app

                    :title "" :desc "" :price 0 :img "" :sold-out false}))

# Как я понял твою концепцию Hero

Ты хочешь сделать **Hero-блок как интерактивный слайдер**, основанный на **4 животных из логотипа UEAVA**.

### Основные элементы:

1️⃣ **Hero состоит из 4 слайдов**

Каждый слайд:

```
фон — изображение животного
(хамелеон / сова / заяц / медведь)

поверх:
заголовок
описание
2 кнопки
```

---

2️⃣ **Переключение слайдов**

Слайды переключаются:

- автоматически
- каждые **5-10 секунд**
- интервал **выбирается случайно**
- следующий слайд тоже **выбирается случайно**

То есть:

```
chameleon
↓
random
↓
owl
↓
random
↓
bear
```

---

3️⃣ **Эффект переключения**

Слайд меняется **слайд-эффектом** (или fade).

Например:

```
slide ←
или
fade
```

---

4️⃣ **Мини-навигация (очень важная часть)**

В одном из углов hero будет **мини-логотип из 4 квадрантов**:

```
┌─────┬─────┐
│ 🦎  │ 🦉  │
├─────┼─────┤
│ 🐇  │ 🐻  │
└─────┴─────┘
```

Это **мини-навигация**.

---

5️⃣ **Поведение мини-навигации**

Она работает так:

### текущий активный слайд

его иконка **подсвечена**

```
┌─────┬─────┐
│ 🦎* │ 🦉  │
├─────┼─────┤
│ 🐇  │ 🐻  │
└─────┴─────┘
```

---

### hover мышкой

если пользователь наводит:

```
hover 🦉
```

то hero **сразу переключается** на этот слайд.

---

6️⃣ **Приоритет управления**

логика примерно такая:

```
auto rotate
↓
user hover
↓
manual override
```

# Почему эта идея хорошая

У тебя **очень сильная бренд-логика**.

Логотип = пазл из животных
Hero = **те же животные**

Это называется **brand continuity**.

Сайт начинает выглядеть **как единая система**.

```
(ns ueava.components.hero
  (:require
   [reagent.core :as r]))

;; ----------------------------
;; состояние
;; ----------------------------

(def current-slide (r/atom 0))
(def timer-id (atom nil))

;; ----------------------------
;; данные слайдов
;; ----------------------------

(def slides
  [{:key :chameleon
    :img "img/Hero_Slide_1_chameleon.webp"
    :title "Why Do You Need A Reptile Vet?"
    :text "Specialized care for exotic animals."
    :btn "Become Member"}

   {:key :owl
    :img "img/Hero_Slide_2_owl.webp"
    :title "Avian Veterinary Excellence"
    :text "Support for bird specialists worldwide."
    :btn "Learn More"}

   {:key :rabbit
    :img "img/Hero_Slide_3_rabbit.webp"
    :title "Small Mammal Medicine"
    :text "Advanced treatment for rabbits and rodents."
    :btn "Join UEAVA"}

   {:key :bear
    :img "img/Hero_Slide_4_bear.webp"
    :title "Wildlife Veterinary Science"
    :text "Supporting vets working with wildlife."
    :btn "Discover"}])

;; ----------------------------
;; random helpers
;; ----------------------------

(defn random-slide []
  (rand-int (count slides)))

(defn random-delay []
  (+ 5000 (rand-int 5000))) ;; 5–10 sec

;; ----------------------------
;; авто переключение
;; ----------------------------

(defn schedule-next! []
  (when @timer-id
    (js/clearTimeout @timer-id))

  (reset! timer-id
          (js/setTimeout
           (fn []
             (reset! current-slide (random-slide))
             (schedule-next!))
           (random-delay))))

;; ----------------------------
;; hero background
;; ----------------------------

(defn background-slider []
  (let [{:keys [img]} (nth slides @current-slide)]
    [:div
     {:class "absolute inset-0 bg-cover bg-center transition-all duration-700"
      :style {:background-image (str "url('" img "')")}}]))

;; ----------------------------
;; hero content
;; ----------------------------

(defn hero-content []
  (let [{:keys [title text btn]} (nth slides @current-slide)]
    [:div
     {:class "relative z-10 text-white text-center max-w-2xl"}

     [:h1 {:class "text-5xl font-bold mb-4"} title]

     [:p {:class "text-xl mb-6"} text]

     [:button
      {:class "bg-white text-black px-6 py-3 rounded-lg"}
      btn]]))

;; ----------------------------
;; mini navigation
;; ----------------------------

(defn nav-item [idx {:keys [img]}]
```

# 1️⃣ Самый простой — вставить SVG прямо как hiccup

Открываешь `logo.svg` и копируешь содержимое внутрь компонента.

SVG:

```xml
<svg viewBox="0 0 200 200">
  <g id="logo-text">...</g>
  <g id="puzzle-1">...</g>
  <g id="puzzle-2">...</g>
</svg>
```

в Reagent превращается в:

```clojure
(defn logo []
  [:svg {:viewBox "0 0 200 200"}
   [:g {:id "logo-text"} ...]
   [:g {:id "puzzle-1"} ...]
   [:g {:id "puzzle-2"} ...]])
```

Теперь можно управлять:

```clojure
[:g {:class (when hide-text "hidden")}]
```

или

```clojure
[:g {:class (when highlight "fill-yellow-400")}]
```

и использовать:


---

## 1️⃣ Импортировать как файл — не для динамики

```clojure
[:> LogoSvg]
```

- Ты сможешь менять **только размеры и цвет всего SVG через CSS**.
- **Нельзя выделять один из пазлов или убрать текст**, потому что все `<path>` зашиты внутри SVG файла.

---

## 2️⃣ Вставить как hiccup (рекомендую)

1. Открываешь SVG.
2. Копируешь `<svg>…</svg>` в `.cljs`.
3. Меняешь все атрибуты из `class="cls-1"` → `:class "fill-ueava-brown"` (или Tailwind классы).
4. Разбиваешь на **компоненты для частей логотипа**:

```clojure
(defn puzzle-1 [] [:path {:d "M75.65,63.47c-..." :class (when highlight? "fill-yellow-400")}])
(defn puzzle-2 [] [:path {:d "M69.88,69.71c-..." :class ""}])
(defn logo []
  [:svg {:viewBox "0 0 1133.86 351.36"}
   [puzzle-1 {:highlight true}]
   [puzzle-2]
   ;; текст можно скрывать через reagent state
   [:text {:class (when hide-text "hidden")} "UEAVA"]])
```

Теперь ты можешь **подсвечивать отдельные пазлы**, скрывать текст или менять части логотипа на лету через `Reagent atoms`.

---

## 3️⃣ Автоматизация конвертации

- Есть онлайн-конвертеры `SVG → React` (который легко превратить в hiccup)
- Потом просто заменяешь `className` на `:class` и правишь атрибуты под Tailwind.
- После этого SVG становится **полностью интерактивным компонентом**.

**** свернуть navigation в hamburger menu
Самое важное для мобильной версии — **свернуть navigation в hamburger menu**. Хорошая новость: это можно сделать **почти без изменений структуры**, используя responsive-классы из Tailwind CSS и маленькое состояние в Reagent.

Я покажу **самый безболезненный вариант** — добавляется примерно **20 строк кода**, а остальная разметка остаётся почти такой же.

---
**** # 1. состояние меню

В Reagent добавим атом:

```clojure
(defonce menu-open? (reagent/atom false))
```

---
**** # 2. кнопка hamburger

Добавим кнопку, которая видна **только на мобильных**.

```clojure
[:button
 {:class "md:hidden text-3xl"
  :on-click #(swap! menu-open? not)}
 "☰"]
```

`md:hidden` означает:

```
mobile → показываем
desktop → скрываем
```

---
**** # 3. navigation для desktop

Твоя текущая навигация почти не меняется:

```clojure
[:nav {:class "hidden md:flex space-x-6 font-medium"}
 [:a {:href "#" :class "hover:text-ueava-sand"} "Home"]
 [:a {:href "#"} "About"]
 [:a {:href "#"} "Resources"]
 [:a {:href "#"} "Membership"]
 [:a {:href "#"} "Conferences"]]
```

`hidden md:flex` означает:

```
mobile → скрыта
desktop → обычное меню
```

---
**** # 4. mobile menu

Теперь добавляем выпадающее меню:

```clojure
(when @menu-open?
  [:div
   {:class "absolute top-full left-0 w-full bg-white shadow-md md:hidden"}

   [:div {:class "flex flex-col p-6 space-y-4"}

    [:a {:href "#"} "Home"]
    [:a {:href "#"} "About"]
    [:a {:href "#"} "Resources"]
    [:a {:href "#"} "Membership"]
    [:a {:href "#"} "Conferences"]]])
```

**** # 5. итоговый header

Получается так:

```clojure
(defn header []
  [:header {:class "relative flex items-center justify-between px-8 py-4 bg-white shadow"}

   ;; logo
   [:img {:src "/img/ueava-logo.svg"
          :class "h-12 w-auto"}]

   ;; desktop nav
   [:nav {:class "hidden md:flex space-x-6 font-medium"}

    [:a {:href "#"} "About"]
    [:a {:href "#"} "Resources"]
    [:a {:href "#"} "Membership"]
    [:a {:href "#"} "Conferences"]]

   ;; hamburger
   [:button
    {:class "md:hidden text-3xl"
     :on-click #(swap! menu-open? not)}
    "☰"]

   ;; mobile menu
   (when @menu-open?
     [:div
      {:class "absolute top-full left-0 w-full bg-white shadow-md md:hidden"}

      [:div {:class "flex flex-col p-6 space-y-4"}
       [:a {:href "#"} "Home"]
       [:a {:href "#"} "About"]
       [:a {:href "#"} "Resources"]
       [:a {:href "#"} "Membership"]
       [:a {:href "#"} "Conferences"]]])])
```

```css
.parallax-slow {
  transform: translateZ(-1px) scale(2);
}

.parallax-medium {
  transform: translateZ(-0.5px) scale(1.5);
}

.parallax-fast {
  transform: translateZ(0);
}
```

И активируем 3D scrolling:

```css
.hero-parallax {
  perspective: 1px;
  overflow-x: hidden;
  overflow-y: auto;
}
```
